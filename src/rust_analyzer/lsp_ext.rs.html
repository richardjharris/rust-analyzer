pub fn supports_utf8(caps: &lsp_types::ClientCapabilities) -> bool {
    match &caps.general {
        Some(general) => general
            .position_encodings
            .as_deref()
            .unwrap_or_default()
            .iter()
            .any(|it| it == &PositionEncodingKind::UTF8),
        _ => false,
    }
}
